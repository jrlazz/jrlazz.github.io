<script type="module">

import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

var scene,camera,renderer,controls;
var cube;
var face=[];
var si=1
var an=Math.PI;

scene=new THREE.Scene();
camera=new THREE.PerspectiveCamera(55,window.innerWidth/window.innerHeight,0.1,1000);
camera.position.set(0,1,5);
renderer=new THREE.WebGLRenderer({antialias:true});
renderer.setSize(window.innerWidth,window.innerHeight);	
document.body.appendChild(renderer.domElement); 

for(let z=1;z<7;z++){face[z]=new THREE.Mesh(new THREE.PlaneGeometry(si,si,si), new THREE.MeshBasicMaterial({color:0xffffff * Math.random(),side:THREE.DoubleSide}));}

face[2].material.transparent=true;
face[2].material.opacity=0.4;

face[2].position.z=si;

face[3].position.x=si/2;
face[3].rotation.y=an/2;
face[3].position.z=si/2;

face[4].position.x=-si/2;
face[4].rotation.y=an/2;
face[4].position.z=si/2;

face[5].position.z=si/2;
face[5].position.y=si/2;
face[5].rotation.x=an/2;

  //face[5].position.z=si/2;
face[6].position.y=-si;
  //face[5]rotation.x=an/2;

cube=new THREE.Object3D();

for(let z=1;z<7;z++){cube.add(face[z]);}

cube.rotation.set(an,an/1.2,an/20);
scene.add(cube);

controls=new OrbitControls(camera,renderer.domElement);

function animate(){
	requestAnimationFrame(animate);
	cube.rotation.y += 0.01;
	renderer.render(scene,camera);
}

animate();

</script>
