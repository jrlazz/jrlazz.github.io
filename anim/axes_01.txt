import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

var renderer,scene,camera,geometry,material,controls,amb;
var group=new THREE.Group();
var boly=[];var coly=[];var geo=[];var mat=[];var moly=[];
var bolygeo,bolymat,colygeo,colymat;

function init(){

	renderer=new THREE.WebGLRenderer({antialias:true,alpha:true});
	renderer.setPixelRatio(window.devicePixelRatio);
	renderer.setSize(window.innerWidth,window.innerHeight);
	renderer.setClearColor('#ccc',1);
	renderer.outputColorSpace=THREE.LinearSRGBColorSpace;

	document.body.appendChild(renderer.domElement);

	scene=new THREE.Scene();

	camera=new THREE.PerspectiveCamera(40,window.innerWidth/window.innerHeight,1,1000);
	camera.position.set(0,0,5);
	
	amb=new THREE.AmbientLight("#fff",6);
	scene.add(amb);

	controls=new OrbitControls(camera, renderer.domElement);

	geo[1]=new THREE.SphereGeometry(0.015,16,16);
	mat[1]=new THREE.MeshStandardMaterial({color:'#000'});
	moly[1]=new THREE.Mesh(geo[1],mat[1]);
	moly[1].geometry.translate(0,0,1);

	geo[2]=new THREE.SphereGeometry(0.015,16,16);
	mat[2]=new THREE.MeshStandardMaterial({color:'#000'});
	moly[2]=new THREE.Mesh(geo[2],mat[2]);
	moly[2].geometry.translate(0,1,0);

	geo[3]=new THREE.SphereGeometry(0.015,16,16);
	mat[3]=new THREE.MeshStandardMaterial({color:'#000'});
	moly[3]=new THREE.Mesh(geo[3],mat[3]);
	moly[3].geometry.translate(1,0,0);

	coly[0]=new THREE.AxesHelper(1);
	coly[0].setColors('#999','#999','#999');

	coly[0].add(moly[1]);
	coly[0].add(moly[2]);
	coly[0].add(moly[3]);

	coly[100]=coly[0].clone();
	coly[100].rotation.set(Math.PI/3,0,0);
	for(let z=1;z<50;z++){coly[100+z]=coly[100].clone();coly[100+z].rotation.set(Math.random()*Math.PI*2,Math.random()*Math.PI*2,Math.random()*Math.PI*2);group.add(coly[100+z]);}

	coly[200]=coly[0].clone();
	coly[200].rotation.set(0,Math.PI/3,0);
	for(let z=1;z<50;z++){coly[200+z]=coly[200].clone();coly[200+z].rotation.set(Math.random()*Math.PI*2,Math.random()*Math.PI*2,Math.random()*Math.PI*2);group.add(coly[200+z]);}

	coly[300]=coly[0].clone();
	coly[300].rotation.set(0,0,Math.PI/3);
	for(let z=1;z<50;z++){coly[300+z]=coly[300].clone();coly[300+z].rotation.set(Math.random()*Math.PI*2,Math.random()*Math.PI*2,Math.random()*Math.PI*2);group.add(coly[300+z]);}

	scene.add(group);
}

function animate(){
	requestAnimationFrame(animate);
	controls.update();
		group.rotation.x +=0.01;
		group.rotation.y +=0.015;
		group.rotation.z +=0.005;
	renderer.render(scene,camera);
}

init();
animate();
